# 实施检查清单生成器 - 使用指南

## 快速开始

### 1. 基础清单生成

```python
from core.checklist_generator import ChecklistGenerator

# 创建生成器
generator = ChecklistGenerator()

# 生成包含所有阶段、所有模块的标准清单
checklist = generator.generate_all_phases_checklist(
    modules=["sourcing", "contract", "buying", "supplier", "spending"],
    version="V2605",
    project_name="我的Ariba实施项目"
)

# 导出为Markdown查看
md_output = generator.export_to_markdown(checklist)
print(md_output)

# 保存到文件
generator.save_to_file(checklist, "checklist.json", "json")
```

### 2. 生成指定阶段和模块的清单

```python
# 只生成需求分析阶段的清单
checklist = generator.generate_checklist(
    phases=["requirements_analysis"],
    modules=["buying", "sourcing"],
    version="V2605"
)
```

### 3. 智能推荐

```python
from recommendation.recommendation_engine import RecommendationEngine, CompanySize, Industry, IntegrationLevel

# 创建推荐引擎
engine = RecommendationEngine()

# 分析项目特征
profile = engine.analyze_project_profile(
    company_size=CompanySize.LARGE_ENTERPRISE.value,
    industry=Industry.MANUFACTURING.value,
    existing_systems=["SAP S/4HANA"],
    integration_level=IntegrationLevel.LEVEL_3.value
)

# 评估风险
risk = engine.assess_risk_level(
    company_size=CompanySize.LARGE_ENTERPRISE.value,
    industry=Industry.MANUFACTURING.value,
    integration_level=IntegrationLevel.LEVEL_3.value,
    has_historical_data=True
)

print(f"风险等级: {risk['risk_level']}")
print(f"风险评分: {risk['risk_score']}/{risk['max_score']}")
```

### 5. 执行追踪

```python

tracker = TrackingEngine()

tracker.update_item_status(

    ItemStatus.IN_PROGRESS.value,
    assignee="张三",
    notes="开始执行"

    checklist["id"],

    ItemStatus.COMPLETED.value

# 获取受阻项
blocked = tracker.get_blocked_items(checklist)

workload = tracker.get_assignee_workload(checklist)

# 生成进度报告
report = tracker.export_progress_report(checklist, "markdown")
```

## 完整工作流程示例

```python
from core.checklist_generator import ChecklistGenerator
from recommendation.recommendation_engine import RecommendationEngine
from template.template_manager import TemplateManager
from tracking.tracking_engine import TrackingEngine, ItemStatus
from analytics.analytics_engine import AnalyticsEngine

# 1. 创建清单
generator = ChecklistGenerator()
checklist = generator.generate_all_phases_checklist(
    modules=["sourcing", "buying", "supplier"],
    version="V2605",
    project_name="XX公司Ariba实施项目"
)

# 2. 智能推荐
recommender = RecommendationEngine()
profile = recommender.analyze_project_profile(
    company_size="large_enterprise",
    industry="manufacturing",
    existing_systems=["SAP S/4HANA"],
    integration_level="level_3"
)
risk = recommender.assess_risk_level(
    company_size="large_enterprise",
    industry="manufacturing",
    integration_level="level_3"
)
recommendations = recommender.recommend_checklist_items(profile, checklist["items"])

# 3. 执行追踪
tracker = TrackingEngine()
for item in checklist["items"][:10]:
    tracker.update_item_status(
        checklist["id"],
        item["id"],
        ItemStatus.COMPLETED.value,
        assignee="实施顾问A"
    )

# 4. 统计分析
analytics = AnalyticsEngine()
stats = analytics.calculate_completion_stats(checklist, tracker.track_records)
health = analytics.calculate_health_score(checklist, tracker.track_records)

# 5. 生成报告
report = analytics.generate_analytics_report(checklist, tracker.track_records)
print(report)
```
