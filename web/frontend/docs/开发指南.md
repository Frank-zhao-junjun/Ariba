## 项目概述

### 项目简介

SAP Ariba 实施助手是一个基于 React + TypeScript 的 Web 应用，为 SAP Ariba 实施项目提供故障排除和清单管理功能。

### 技术栈

| 层级 | 技术 | 版本 |
|-----|------|------|
| 前端框架 | React | 18.x |
| 类型系统 | TypeScript | 5.x |
| UI 组件库 | Ant Design | 5.x |
| 图表库 | ECharts | 5.x |
| 状态管理 | React Query | 5.x |
| 路由 | React Router | 6.x |
| 构建工具 | Vite | 5.x |
| 后端框架 | FastAPI | 0.109+ |
| Python | Python | 3.10+ |

### API 调用规范

```typescript
// 1. 定义类型
import type { KnowledgeItem } from '@/types'

// 2. 定义 API 函数
export const myApi = {
  getData: (params: QueryParams) => 
    api.get<BaseResponse<DataType[]>>('/endpoint', { params }),
  
  createData: (data: CreateData) => 
    api.post<BaseResponse<DataType>>('/endpoint', data),
  
  updateData: (id: string, data: UpdateData) => 
    api.patch<BaseResponse<DataType>>(`/endpoint/${id}`, data),
  
  deleteData: (id: string) => 
    api.delete<BaseResponse<void>>(`/endpoint/${id}`),
}

// 3. 在组件中使用 React Query
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query'

function MyComponent() {
  const queryClient = useQueryClient()
  
  // 查询
  const { data, isLoading } = useQuery({
    queryKey: ['my-data'],
    queryFn: () => myApi.getData(params),
    enabled: true, // 条件执行
    staleTime: 5 * 60 * 1000, // 5分钟缓存
  })
  
  // 变更
  const mutation = useMutation({
    mutationFn: myApi.createData,
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['my-data'] })
    },
  })
  
  // ...
}
```

## 测试指南

### 单元测试

```typescript
// src/components/__tests__/MyComponent.test.tsx
import { render, screen, fireEvent } from '@testing-library/react'
import MyComponent from '../MyComponent'

describe('MyComponent', () => {
  it('renders title', () => {
    render(<MyComponent title="Test Title" />)
    expect(screen.getByText('Test Title')).toBeInTheDocument()
  })

  it('handles click', () => {
    const onClick = jest.fn()
    render(<MyComponent title="Test" onClick={onClick} />)
    fireEvent.click(screen.getByRole('button'))
    expect(onClick).toHaveBeenCalled()
  })
})
```

### 集成测试

```typescript
// tests/integration/api.test.ts
import { renderHook, waitFor } from '@testing-library/react'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { useMyData } from '@/hooks/useMyData'

const createWrapper = () => {
  const queryClient = new QueryClient()
  return ({ children }: any) => (
    <QueryClientProvider client={queryClient}>
      {children}
    </QueryClientProvider>
  )
}

describe('useMyData', () => {
  it('fetches data successfully', async () => {
    const { result } = renderHook(() => useMyData(), {
      wrapper: createWrapper(),
    })

    await waitFor(() => expect(result.current.isSuccess).toBe(true))
    expect(result.current.data).toBeDefined()
  })
})
```

## 部署上线

### 前端部署

#### 构建生产版本

```bash
# 安装依赖
npm install

# 构建
npm run build

# 预览
npm run preview
```

#### 部署配置

```typescript
// vite.config.ts
export default defineConfig({
  base: '/',  // 或 '/sub-path/'
  build: {
    outDir: 'dist',
    sourcemap: false,
    rollupOptions: {
      output: {
        manualChunks: {
          vendor: ['react', 'react-dom', 'react-router-dom'],
          antd: ['antd'],
        },
      },
    },
  },
})
```

### 后端部署

#### 使用 Docker

```dockerfile
# Dockerfile
FROM python:3.10-slim

WORKDIR /app
COPY requirements.txt .
RUN pip install -r requirements.txt

COPY . .
CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
```

#### 运行服务

```bash
# 开发环境
uvicorn main:app --reload

# 生产环境
uvicorn main:app --host 0.0.0.0 --port 8000 --workers 4
```

### Nginx 配置

```nginx
server {
    listen 80;
    server_name example.com;
    
    # 前端静态文件
    location / {
        root /var/www/app;
        try_files $uri $uri/ /index.html;

    location /api {
        proxy_pass http://localhost:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
```
