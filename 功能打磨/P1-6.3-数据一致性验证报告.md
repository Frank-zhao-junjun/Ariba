## 数据一致性风险点

### 1. 客户端状态一致性

| 风险点 | 描述 | 影响 |
|-------|------|------|
| 缓存与服务器数据不一致 | React Query缓存数据与服务器不同步 | 数据显示错误 |
| 乐观更新后服务器返回不同结果 | 乐观更新与实际服务器结果冲突 | 数据状态异常 |
| 多个组件共享数据不一致 | 不同组件显示同一数据不一致 | 用户困惑 |

### 2. 跨组件数据同步

| 风险点 | 描述 | 影响 |
|-------|------|------|
| 列表与详情数据不一致 | 列表显示的数据与详情页不同 | 数据陈旧 |
| 父子组件状态不同步 | 父组件状态与子组件状态不一致 | 逻辑错误 |
| 模态框与表单数据不同步 | 模态框显示的数据与表单数据不一致 | 操作错误 |

### 3. 表单数据一致性

| 风险点 | 描述 | 影响 |
|-------|------|------|
| 表单验证与提交数据不一致 | 验证通过但提交失败 | 用户体验差 |
| 表单重置不完整 | 部分字段未重置 | 数据残留 |
| 多步骤表单数据丢失 | 步骤间数据未正确传递 | 操作失败 |

## 数据一致性保障机制

### 1. 缓存与服务器同步

```typescript
// React Query缓存同步机制
const queryClient = useQueryClient()

// 手动使缓存失效
const handleDataChange = async () => {
  // 更新服务器数据
  await updateServerData()
  
  // 立即使相关缓存失效，确保下次请求获取最新数据
  queryClient.invalidateQueries({
    queryKey: ['projects'],
    refetchType: 'all', // 所有相关查询都重新获取
  })
}

// 或者直接更新缓存
queryClient.setQueryData(['project', id], (old) => ({
  ...old,
  updatedField: newValue,
}))
```

### 2. 乐观更新与服务器确认

```typescript
// 乐观更新确保即时响应
const mutation = useMutation({
  mutationFn: updateProject,
  
  // 更新前的处理
  onMutate: async (newData) => {
    // 取消任何正在进行的请求，避免覆盖乐观更新
    await queryClient.cancelQueries(['project', newData.id])
    
    // 保存当前值，用于回滚
    const previousData = queryClient.getQueryData(['project', newData.id])
    
    // 立即更新缓存
    queryClient.setQueryData(['project', newData.id], (old) => ({
      ...old,
      ...newData,
    }))
    
    return { previousData }
  },
  
  // 更新失败时的回滚
  onError: (err, newData, context) => {
    // 恢复到之前的值
    queryClient.setQueryData(['project', newData.id], context.previousData)
    
    // 显示错误消息
    message.error('更新失败，已恢复原状态')
  },
  
  // 无论成功失败都执行
  onSettled: () => {
    // 重新获取最新数据以确保一致
    queryClient.invalidateQueries(['project', id])
  },
})
```

### 3. 跨组件数据同步

```typescript
// 使用Context共享数据并保证一致性
const DataContext = createContext<DataContextType>(null!)

function DataProvider({ children }) {
  const queryClient = useQueryClient()
  
  // 统一的数据更新方法
  const updateData = useCallback((key: string, data: any) => {
    // 1. 更新服务器
    return api.update(key, data)
      .then((response) => {
        // 2. 同步更新所有相关缓存
        queryClient.setQueryData([key], response.data)
        return response.data
      })
      .catch((error) => {
        // 3. 失败时使缓存失效
        queryClient.invalidateQueries([key])
        throw error
      })
  }, [queryClient])
  
  return (
    <DataContext.Provider value={{ updateData }}>
      {children}
    </DataContext.Provider>
  )
}

// 所有组件使用同一Provider，确保数据一致
function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <DataProvider>
        <Router />
      </DataProvider>
    </QueryClientProvider>
  )
}
```

// 前端数据验证
const validateProject = (data: Partial<Project>): ValidationResult => {
  const errors: Record<string, string> = {}

  // 必填字段验证
  if (!data.name?.trim()) {
    errors.name = '项目名称不能为空'
  }

  if (!data.version) {
    errors.version = '请选择版本'
  }

  // 格式验证
  if (data.name && data.name.length > 100) {
    errors.name = '项目名称不能超过100个字符'
  }

  // 业务规则验证
  if (data.modules && data.modules.length === 0) {
    errors.modules = '请至少选择一个模块'
  }

  return {
    valid: Object.keys(errors).length === 0,
    errors,
  }
}

// 使用验证
const handleSubmit = async (data) => {
  const result = validateProject(data)
  
  if (!result.valid) {
    return { success: false, errors: result.errors }
  }
  
  return api.submitProject(data)
}

## 数据一致性最佳实践

### 1. 单点数据来源

```typescript
// 一个数据源，多个消费者
// 使用React Query作为唯一数据源
const useProject = (id: string) => {
  const queryClient = useQueryClient()
  
  const query = useQuery({
    queryKey: ['project', id],
    queryFn: () => api.getProject(id),
    // 关键：让React Query成为唯一数据源
  })
  
  // 通过Query Client更新
  const updateProject = async (data) => {
    await queryClient.fetchQuery({
      queryKey: ['project', id],
      queryFn: () => api.updateProject(id, data),
    })
  }
  
  return { ...query, updateProject }
}
```

### 2. 规范化数据流

```typescript
// 统一的数据更新流程
// 1. 乐观更新UI
// 2. 发送到服务器
// 3. 成功则确认，失败则回滚
// 4. 最后同步最新数据

async function dataUpdateFlow(queryKey, updateData) {
  const previousData = queryClient.getQueryData(queryKey)
  
  // 步骤1: 乐观更新
  queryClient.setQueryData(queryKey, (old) => ({ ...old, ...updateData }))
  
  try {
    // 步骤2: 发送请求
    await api.update(queryKey, updateData)
    
    // 步骤3: 使缓存失效，获取最新数据
    await queryClient.invalidateQueries(queryKey)
  } catch (error) {
    // 步骤4: 回滚
    queryClient.setQueryData(queryKey, previousData)
    throw error
  }
}
```

### 3. 错误边界保护

```typescript
// 错误边界防止状态不一致导致应用崩溃
class DataBoundary extends Component {
  state = { hasError: false, error: null }

  static getDerivedStateFromError(error) {
    return { hasError: true, error }
  }

  componentDidCatch(error, errorInfo) {
    console.error('Data error:', error, errorInfo)
    // 可以在这里尝试恢复数据一致性
  }

  render() {
    if (this.state.hasError) {
      return (
        <ErrorFallback
          error={this.state.error}
          onRetry={() => {
            this.setState({ hasError: false })
          }}
        />
      )
    }
    return this.props.children
  }
}
```
