### 2. 请求取消 (AbortController)

```typescript
// 请求取消机制
export const useCancellableQuery = (url: string) => {
  const [data, setData] = useState(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    const controller = new AbortController()

    const fetchData = async () => {
      try {
        const res = await fetch(url, { signal: controller.signal })
        const json = await res.json()
        setData(json)
      } catch (err) {
        if (err.name !== 'AbortError') {
          console.error('Fetch error:', err)
        }
      } finally {
        setLoading(false)
      }
    }

    fetchData()

    // 组件卸载时取消请求
    return () => controller.abort()
  }, [url])

  return { data, loading }
}
```

### 3. 乐观更新与回滚

```typescript
// 乐观更新Hook
export const useOptimisticUpdate = () => {
  const queryClient = useQueryClient()

  const optimisticUpdate = async ({
    queryKey,
    updateFn,
    mutationFn,
    onError,
  }: OptimisticUpdateParams) => {
    // 1. 保存旧值
    const previousData = queryClient.getQueryData(queryKey)

    // 2. 乐观更新
    queryClient.setQueryData(queryKey, (old) => updateFn(old))

    try {
      // 3. 执行实际更新
      await mutationFn()
    } catch (error) {
      // 4. 失败时回滚
      queryClient.setQueryData(queryKey, previousData)
      onError?.(error)
    }
  }

  return { optimisticUpdate }
}

// 使用示例
const { optimisticUpdate } = useOptimisticUpdate()

optimisticUpdate({
  queryKey: ['project', projectId],
  updateFn: (old) => ({
    ...old,
    status: 'completed',
  }),
  mutationFn: () => updateProject(projectId, { status: 'completed' }),
  onError: (err) => {
    message.error('更新失败，已恢复原状态')
  },
})
```

### 4. 竞态条件处理

```typescript
// 使用请求ID处理竞态条件
let requestId = 0

export const useRaceRequest = () => {
  const [data, setData] = useState(null)

  const fetchWithRace = async (url: string) => {
    const currentId = ++requestId
    
    try {
      const res = await fetch(url)
      const json = await res.json()
      
      // 只处理最新请求的结果
      if (currentId === requestId) {
        setData(json)
      }
    } catch (err) {
      if (currentId === requestId) {
        setData(null)
      }
    }
  }

  return { data, fetchWithRace }
}
```

// 用于搜索输入等高频触发场景
import { useDebouncedCallback } from 'use-debounce'

function SearchInput() {
  const [value, setValue] = useState('')
  
  const debouncedSearch = useDebouncedCallback(
    (searchTerm) => {
      // 执行搜索
      console.log('Searching for:', searchTerm)

  return (
    <Input
      value={value}

        setValue(e.target.value)
        debouncedSearch(e.target.value)
      }}
    />
  )
}

### 2. 节流 (Throttle)

```typescript
// 用于滚动事件等高频触发场景
import { useThrottledCallback } from 'use-debounce'

function ScrollTracker() {
  const [scrollY, setScrollY] = useState(0)

  const throttledScroll = useThrottledCallback(
    () => {
      setScrollY(window.scrollY)
    },
    100 // 100ms节流
  )

  useEffect(() => {
    window.addEventListener('scroll', throttledScroll)
    return () => window.removeEventListener('scroll', throttledScroll)
  }, [throttledScroll])

  return <div>Scroll: {scrollY}px</div>
}
```

### 3. 并发限制 (Semaphore)

```typescript
// 控制同时执行的最大并发数
class Semaphore {
  private permits: number
  private queue: (() => void)[] = []

  constructor(permits: number) {
    this.permits = permits
  }

  async acquire(): Promise<void> {
    if (this.permits > 0) {
      this.permits--
      return Promise.resolve()
    }

    return new Promise((resolve) => {
      this.queue.push(resolve)
    })
  }

  release(): void {
    this.permits++
    const next = this.queue.shift()
    if (next) {
      this.permits--
      next()
    }
  }
}

// 使用示例
const semaphore = new Semaphore(3) // 最多3个并发

async function fetchWithLimit(url: string) {
  await semaphore.acquire()
  try {
    return await fetch(url)
  } finally {
    semaphore.release()
  }
}
```
