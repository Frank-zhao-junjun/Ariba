```typescript
// React Query重试配置
const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      // 失败后重试3次
      retry: 3,
      // 重试延迟：指数退避
      retryDelay: (attemptIndex) => Math.min(1000 * 2 ** attemptIndex, 30000),
      // 网络错误才重试
      retryWhen: (error) => error instanceof NetworkError,
    },
    mutations: {
      // 失败后重试1次
      retry: 1,
      retryDelay: 1000,
    },
  },
})

// 自定义重试逻辑
const useRetryQuery = (queryKey, queryFn, options = {}) => {
  const [retryCount, setRetryCount] = useState(0)
  const maxRetries = options.maxRetries || 3

  const fetchWithRetry = async () => {
    try {
      return await queryFn()
    } catch (error) {
      if (retryCount < maxRetries) {
        setRetryCount((c) => c + 1)
        // 指数退避
        const delay = Math.min(1000 * 2 ** retryCount, 10000)
        await new Promise((resolve) => setTimeout(resolve, delay))
        return fetchWithRetry()
      }
      throw error
    }
  }

  return useQuery(queryKey, fetchWithRetry)
}
```

### 2. 超时处理

```typescript
// 请求超时配置
const API_TIMEOUT = 10000 // 10秒

const fetchWithTimeout = async (url: string, options = {}) => {
  const controller = new AbortController()
  const timeoutId = setTimeout(() => controller.abort(), API_TIMEOUT)

  try {
    const response = await fetch(url, {
      ...options,
      signal: controller.signal,
    })
    clearTimeout(timeoutId)
    return response
  } catch (error) {
    clearTimeout(timeoutId)
    if (error.name === 'AbortError') {
      throw new Error('请求超时，请稍后重试')
    }
    throw error
  }
}

// React Query超时配置
const { data, isError, error } = useQuery({
  queryKey: ['data'],
  queryFn: () => fetchWithTimeout('/api/data'),
  // 超时后5秒重试
  retry: (failureCount, error) => {
    if (error.message.includes('超时')) {
      return failureCount < 2
    }
    return failureCount < 3
  },
})
```

### 3. 降级处理

```typescript
// 服务降级Hook
const useFallbackData = <T>({
  queryKey,
  queryFn,
  fallbackData,
  fallbackDuration = 60000, // 降级数据有效期60秒
}: UseFallbackOptions<T>) => {
  const { data, isError, dataUpdatedAt } = useQuery({
    queryKey,
    queryFn,
  })

  // 检查降级数据是否过期
  const isFallbackExpired = Date.now() - dataUpdatedAt > fallbackDuration

  // 如果当前数据不可用且降级数据未过期，返回降级数据
  if (isError || isFallbackExpired) {
    return {
      data: fallbackData,
      isLoading: false,
      isFallback: true,
    }
  }

  return {
    data,
    isLoading: false,
    isFallback: false,
  }
}

// 使用示例
const { data, isFallback } = useFallbackData({

  fallbackData: defaultStatistics,
})

return (

    <StatisticsChart data={data} />
  </div>
```

### 4. 离线支持

```typescript
// 离线操作队列
class OfflineQueue {
  private queue: Operation[] = []
  private isOnline = navigator.onLine

  constructor() {
    window.addEventListener('online', () => this.processQueue())
    window.addEventListener('offline', () => this.handleOffline())
  }

  add(operation: Operation) {
    this.queue.push(operation)
    this.persistToStorage()
  }

  async processQueue() {
    if (!this.isOnline || this.queue.length === 0) return

    const failed: Operation[] = []

    for (const operation of this.queue) {
      try {
        await this.executeOperation(operation)
      } catch (error) {
        failed.push(operation)
      }
    }

    this.queue = failed
    this.persistToStorage()
  }

  private persistToStorage() {
    localStorage.setItem('offline-queue', JSON.stringify(this.queue))
  }

  private handleOffline() {
    this.isOnline = false
    console.log('网络已断开，操作已保存到队列')
  }
}

// 离线操作Hook
const useOfflineOperation = () => {
  const [isOnline, setIsOnline] = useState(navigator.onLine)
  const [pendingCount, setPendingCount] = useState(0)
  const queue = useMemo(() => new OfflineQueue(), [])

  useEffect(() => {
    const handleOnline = () => {
      setIsOnline(true)
      setPendingCount(0)
    }
    const handleOffline = () => setIsOnline(false)

    window.addEventListener('online', handleOnline)
    window.addEventListener('offline', handleOffline)

    return () => {
      window.removeEventListener('online', handleOnline)
      window.removeEventListener('offline', handleOffline)
    }
  }, [])

  const addToQueue = (operation: Operation) => {
    if (!isOnline) {
      queue.add(operation)
      setPendingCount((c) => c + 1)
    }
  }

  return { isOnline, pendingCount, addToQueue }
}
```

```typescript
// 全局错误边界
class GlobalErrorBoundary extends Component {
  state = { hasError: false, error: null }

  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, errorInfo: ErrorInfo) {
    console.error('Global error:', error, errorInfo)
    
    // 可以在这里上报错误日志
    logError(error, errorInfo)
  }

  handleRetry = () => {
    this.setState({ hasError: false, error: null })
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="error-boundary">
          <ErrorIcon />
          <h2>出错了</h2>
          <p>{this.state.error?.message}</p>
          <Button onClick={this.handleRetry}>重试</Button>
          <Button onClick={() => window.location.href = '/'}>返回首页</Button>
        </div>
      )
    }
    return this.props.children
  }
}

// 页面级错误边界
class PageErrorBoundary extends Component {
  state = { hasError: false }

  render() {
    if (this.state.hasError) {
      return <PageErrorFallback onRetry={() => window.location.reload()} />
    }
    return this.props.children
  }
}
```

### 2. 状态恢复

```typescript
// 状态持久化和恢复
const STORAGE_KEY = 'app-state'

interface PersistedState {
  projectFilters: FilterState
  userPreferences: UserPreferences
  recentSearches: string[]
}

// 保存状态
const persistState = (state: Partial<PersistedState>) => {
  const current = loadState()
  const updated = { ...current, ...state }
  localStorage.setItem(STORAGE_KEY, JSON.stringify(updated))
}

// 加载状态
const loadState = (): PersistedState => {
  try {
    const saved = localStorage.getItem(STORAGE_KEY)
    return saved ? JSON.parse(saved) : getDefaultState()
  } catch {
    return getDefaultState()
  }
}

// 恢复状态Hook
const usePersistedState = () => {
  const [state, setState] = useState<PersistedState>(() => loadState())

  // 状态变化时自动保存
  useEffect(() => {
    persistState(state)
  }, [state])

  // 清除状态
  const clearState = () => {
    localStorage.removeItem(STORAGE_KEY)
    setState(getDefaultState())
  }

  return { state, setState, clearState }
}
```

### 3. 回滚机制

```typescript
// 操作历史记录
class OperationHistory {
  private history: Operation[] = []
  private currentIndex = -1
  private maxSize = 50

  push(operation: Operation) {
    // 删除当前位置之后的历史
    this.history = this.history.slice(0, this.currentIndex + 1)
    this.history.push(operation)
    
    // 限制历史大小
    if (this.history.length > this.maxSize) {
      this.history.shift()
    } else {
      this.currentIndex++
    }
  }

  undo(): Operation | null {
    if (this.currentIndex >= 0) {
      const operation = this.history[this.currentIndex]
      this.currentIndex--
      return operation
    }
    return null
  }

  redo(): Operation | null {
    if (this.currentIndex < this.history.length - 1) {
      this.currentIndex++
      return this.history[this.currentIndex]
    }
    return null
  }

  canUndo() { return this.currentIndex >= 0 }
  canRedo() { return this.currentIndex < this.history.length - 1 }
}

// 使用回滚Hook
const useUndoRedo = () => {
  const history = useMemo(() => new OperationHistory(), [])
  const [canUndo, setCanUndo] = useState(false)
  const [canRedo, setCanRedo] = useState(false)

  const update = (operation: Operation) => {
    history.push(operation)
    setCanUndo(history.canUndo())
    setCanRedo(history.canRedo())
  }

  const undo = async () => {
    const operation = history.undo()
    if (operation) {
      await executeReverse(operation)
      setCanUndo(history.canUndo())
      setCanRedo(history.canRedo())
    }
  }

  const redo = async () => {
    const operation = history.redo()
    if (operation) {
      await executeOperation(operation)
      setCanUndo(history.canUndo())
      setCanRedo(history.canRedo())
    }
  }

  return { update, undo, redo, canUndo, canRedo }
}
```
