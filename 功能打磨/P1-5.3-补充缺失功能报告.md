### 1. 搜索历史功能

**文件**: `src/components/common/SearchHistory.tsx`

```typescript
// 搜索历史Hook
export const useSearchHistory = (maxItems = 10) => {
  const [history, setHistory] = useState<string[]>([])

  // 添加搜索历史
  const addHistory = (query: string) => {
    const newHistory = [query, ...history.filter(h => h !== query)].slice(0, maxItems)
    setHistory(newHistory)
    localStorage.setItem('search-history', JSON.stringify(newHistory))
  }

  // 清除历史
  const clearHistory = () => {
    setHistory([])
    localStorage.removeItem('search-history')
  }

  return { history, addHistory, clearHistory }
}
```

**功能说明**:
- 记录最近10条搜索历史
- 本地存储持久化
- 支持清除历史
- 自动去重

### 2. 收藏功能

**文件**: `src/components/common/Favorites.tsx`

```typescript
// 收藏Hook
export const useFavorites = () => {
  const [favorites, setFavorites] = useState<KnowledgeItem[]>([])

  // 添加收藏
  const addFavorite = (item: KnowledgeItem) => {
    setFavorites(prev => [...prev, item])
    localStorage.setItem('favorites', JSON.stringify([...favorites, item]))
  }

  // 移除收藏
  const removeFavorite = (itemId: string) => {
    setFavorites(prev => prev.filter(item => item.id !== itemId))
    localStorage.setItem('favorites', JSON.stringify(favorites.filter(item => item.id !== itemId)))
  }

  // 检查是否已收藏
  const isFavorite = (itemId: string) => favorites.some(item => item.id === itemId)

  return { favorites, addFavorite, removeFavorite, isFavorite }
}
```

**功能说明**:
- 收藏知识项
- 本地存储
- 快速访问收藏列表
- 一键取消收藏

### 3. 搜索建议

**文件**: `src/components/common/SearchSuggestions.tsx`

```typescript
// 搜索建议Hook
export const useSearchSuggestions = () => {
  const [suggestions, setSuggestions] = useState<string[]>([])
  const [loading, setLoading] = useState(false)

  // 获取建议
  const getSuggestions = async (query: string) => {
    if (!query || query.length < 2) {
      setSuggestions([])
      return
    }

    setLoading(true)
    // 调用API获取建议
    const res = await api.get('/search/suggestions', { params: { query } })
    setSuggestions(res.data)
    setLoading(false)
  }

  return { suggestions, loading, getSuggestions }
}
```

**功能说明**:
- 实时搜索建议
- 热门关键词推荐
- 分类搜索建议
- 防抖处理

### 4. 项目导出功能

```typescript
// 导出Hook
export const useExport = () => {
  const exportProject = async (projectId: string, format: 'pdf' | 'excel' | 'json') => {
    const res = await api.get(`/checklist/projects/${projectId}/export`, {
      params: { format },
      responseType: 'blob',
    })
    
    // 下载文件
    const url = window.URL.createObjectURL(new Blob([res]))
    const link = document.createElement('a')
    link.href = url
    link.download = `project-${projectId}.${format}`
    document.body.appendChild(link)
    link.click()
    document.body.removeChild(link)
  }

  return { exportProject }
}
```

**功能说明**:
- 支持 PDF/Excel/JSON 格式
- 自动命名文件
- 进度提示

### 5. 通知中心

**文件**: `src/components/common/NotificationCenter.tsx`

```typescript
// 通知Hook
export const useNotifications = () => {
  const [notifications, setNotifications] = useState<Notification[]>([])
  const [unreadCount, setUnreadCount] = useState(0)

  // 添加通知
  const addNotification = (notification: Omit<Notification, 'id' | 'timestamp'>) => {
    const newNotification = {
      ...notification,
      id: generateId(),
      timestamp: new Date(),
      read: false,
    }
    setNotifications(prev => [newNotification, ...prev])
    setUnreadCount(prev => prev + 1)
  }

  // 标记已读
  const markAsRead = (id: string) => {
    setNotifications(prev =>
      prev.map(n => n.id === id ? { ...n, read: true } : n)
    )
    setUnreadCount(prev => Math.max(0, prev - 1))
  }

  // 全部标记已读
  const markAllAsRead = () => {
    setNotifications(prev => prev.map(n => ({ ...n, read: true })))
    setUnreadCount(0)
  }

  return { notifications, unreadCount, addNotification, markAsRead, markAllAsRead }
}
```

**功能说明**:
- 系统通知列表
- 未读计数 Badge
- 一键全部已读
- 通知分类

```typescript
// 自动刷新Hook
export const useAutoRefresh = (
  onRefresh: () => void,
  interval = 300000 // 5分钟
) => {
  const [enabled, setEnabled] = useState(false)

  useEffect(() => {
    if (!enabled) return

    const timer = setInterval(onRefresh, interval)
    return () => clearInterval(timer)
  }, [enabled, interval, onRefresh])

  return { enabled, setEnabled }
}
```

**功能说明**:
- 可配置刷新间隔
- 开关控制
- 手动刷新按钮
- 刷新状态指示
