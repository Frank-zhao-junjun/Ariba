## 技术亮点

### 1. 组件化设计
```typescript
// 所有组件支持按需导入
import { ContentLoading, SearchEmpty, ErrorDisplay } from '@/components/common'

// 或统一导入
import { CommonComponents } from '@/components/common'
const { ContentLoading, SearchEmpty } = CommonComponents
```

### 2. Hook模式
```typescript
// 通知Hook
const { showSuccess, showError } = useNotification()

// 快捷键Hook
const { shortcuts, flatShortcuts } = useQuickActions(navigate)
useGlobalShortcuts(flatShortcuts)
```

### 3. 配置化设计
```typescript
// 错误配置
const errorConfigs: Record<ErrorType, ErrorConfig> = {
  [ErrorType.NETWORK]: { ... },
  [ErrorType.SERVER]: { ... },
}

// 快捷键配置
const shortcuts: ShortcutCategory[] = [
  { title: '导航', shortcuts: [...] },
  { title: '操作', shortcuts: [...] },
]
```

### 4. 平台适配
```typescript
// Mac显示⌘，Windows显示Ctrl
parts.push(navigator.platform.includes('Mac') ? '⌘' : 'Ctrl')
```
