## 技术实现

### Hook使用示例
```typescript
// 在组件中使用快捷键
import { useGlobalShortcuts } from '@/components/common'

function MyComponent() {
  const shortcuts = [
    {
      key: 'n',
      modifiers: ['ctrl'],
      description: '新建项目',
      action: () => navigate('/projects/new'),
    },
    {
      key: '/',
      description: '聚焦搜索',
      action: () => searchInput.focus(),
    },
  ]

  useGlobalShortcuts(shortcuts)
  
  return <div>...</div>
}
```

### 快捷键格式化
```typescript
import { formatShortcut } from '@/components/common'

// Mac: ⌘ + ⇧ + N
// Windows: Ctrl + Shift + N
formatShortcut({
  key: 'n',
  modifiers: ['ctrl', 'shift'],
  description: '新建项目',
  action: () => {},
})
```

### 快捷键说明弹窗

import { ShortcutHelpModal } from '@/components/common'

const categories = [
