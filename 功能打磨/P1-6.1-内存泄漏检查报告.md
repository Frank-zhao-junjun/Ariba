```typescript
// ❌ 错误：每次渲染创建新的函数
function BadComponent() {
  useEffect(() => {
    const handler = () => {} // 每次渲染都创建新的handler
    window.addEventListener('resize', handler)
  }, []) // 依赖数组为空，但handler是新的引用
}

// ✅ 正确：使用useCallback缓存函数
function GoodComponent() {
  const handler = useCallback(() => {}, [])
  useEffect(() => {
    window.addEventListener('resize', handler)
    return () => window.removeEventListener('resize', handler)
  }, [handler])
}
```

## 已识别的潜在泄漏点

### 1. 定时器泄漏

**问题**: setInterval未在组件卸载时清除

**修复方案**:
```typescript
// ✅ 修复后
useEffect(() => {
  const timer = setInterval(() => {
    fetchData()
  }, 30000)

  return () => clearInterval(timer) // 清理定时器
}, [])
```

### 2. 事件监听器泄漏

**问题**: window事件监听未移除

**修复方案**:
```typescript
// ✅ 修复后
useEffect(() => {
  const handleResize = () => updateDimensions()
  const handleScroll = () => updateScroll()
  
  window.addEventListener('resize', handleResize)
  window.addEventListener('scroll', handleScroll)

  return () => {
    window.removeEventListener('resize', handleResize)
    window.removeEventListener('scroll', handleScroll)
  }
}, [])
```

### 3. React Query缓存

**问题**: 长时间运行可能积累旧数据

**修复方案**:
```typescript
// 配置缓存策略
const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 5 * 60 * 1000, // 5分钟后变为stale
      gcTime: 30 * 60 * 1000, // 30分钟后垃圾回收
      refetchOnWindowFocus: false,
    },
  },
})
```

## 内存优化最佳实践

### 1. 组件级别

```typescript
// 使用useRef存储不需要触发重渲染的值
const intervalRef = useRef<NodeJS.Timeout>()

useEffect(() => {
  intervalRef.current = setInterval(fetchData, 30000)
  return () => clearInterval(intervalRef.current)
}, [])

// 避免在渲染时创建新对象
const memoizedValue = useMemo(() => expensiveCalculation(data), [data])
const memoizedCallback = useCallback(() => doSomething(data), [data])
```

### 2. 列表渲染优化

```typescript
// 使用React.memo避免不必要的重渲染
const ListItem = React.memo(({ item }) => (
  <div>{item.name}</div>
))

// 使用虚拟列表处理大量数据
import { FixedSizeList } from 'react-window'
```

### 3. 图片和媒体优化

```typescript
// 使用懒加载
const LazyImage = ({ src }) => {
  const [loaded, setLoaded] = useState(false)
  
  return (
    <img
      src={src}
      onLoad={() => setLoaded(true)}
      style={{ opacity: loaded ? 1 : 0, transition: 'opacity 0.3s' }}
    />
  )
}
```

### 4. 本地存储清理

```typescript
// 组件卸载时清理localStorage
useEffect(() => {
  const handleBeforeUnload = () => {
    // 清理临时数据
    localStorage.removeItem('temp-data')
  }
  
  window.addEventListener('beforeunload', handleBeforeUnload)
  return () => window.removeEventListener('beforeunload', handleBeforeUnload)
}, [])
```
